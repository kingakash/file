   int binarySearch(int a[],int f,int l,int x)
  
   
  { int mid;
      while (f <= l) {
      mid =(f+l)/2;
      if (a[mid] < x)
         f = mid + 1;    
      else if (a[mid] == x) {
       return mid;
         break;
      }
      else
         l = mid - 1;
      
   }
